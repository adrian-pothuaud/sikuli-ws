'''
This module is intended to demonstrate how to manipulate paths for Sikuli
e.g Manipulate Image Path
e.g Manipulate execution Path
'''

# import sikuli features
from sikuli import *

import sys

# get actual execution path
# also use pathUtils.printBundle() in allos/pathUtils.sikuli
for p in sys.path:
    print(p)

# actual path is [ROOT]/.../paths.sikuli
print(getBundlePath())

# add a path to sys.path
newPath = os.path.dirname(os.path.dirname(getBundlePath()))
if not newPath in sys.path:
    sys.path.append(newPath)
